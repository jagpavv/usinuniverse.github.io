### 기능 구현

* 디자인 가이드, 기획없이 시작되었으며 컨셉은 포인트 리워드를 기반으로한 채팅 어플.
* 회원가입, 로그인이 가능합니다. (전화번호, 이메일 주소 등 서버에서 유효값 체크)
* 내 프로필 수정이 가능합니다.
* 친구의 프로필 확인이 가능합니다.
* 사진은 gif, webp 지원합니다.
* 이메일 주소, 전화번호로 친구추가가 가능합니다.
* 초대장을 전송할 수 있습니다.
* 일대일, 일대다 채팅이 가능합니다.
* 메시지를 읽지 않은 사용자수를 알 수 있습니다.
* 채팅방 프로필 수정과 확인이 가능합니다.
* 채팅중 새로운 친구를 초대할 수 있습니다.
* 처음 채팅방 진입시 30개 이하의 메시지만 볼 수 있으며, Pull To Refresh로 이전 메시지를 확인할 수 있습니다.
* 메시지는 Realm을 통해 저장되며 로그아웃시 모든 저장된 데이터가 삭제됩니다.
* 이모티콘을 주고 받을 수 있습니다.
* 이모티콘 샵에서 새로운 이모티콘을 다운로드할 수 있습니다.
* 이모티콘을 전송하거나 받는 경우 1 포인트를 획득합니다.
* 이모티콘 옆 광고이미지를 클릭하면 3 포인트를 획득합니다.
* 광고이미지를 클릭해 광고 홈페이지로 이동하면 5 포인트를 획득합니다.
* 채팅방에서 획득한 포인트와 내가 보유한 전체 포인트를 실시간으로 확인 가능합니다.
* 설정에서 알림을 on off 할 수 있습니다.

* ChatModelManager

```swift
import Foundation

import RealmSwift



/// 채팅 내역 관리자
struct ChatModelManager {
    // MARK: - Properties
    
    private let realm = try! Realm()
    
    
    
    // MARK: - Methods
    
    /// 채팅 내역 DB에 저장하기
    ///
    /// - parameters:
    ///     - message: Stringfy된 JSON 데이터
    ///     - roomID: 채팅 내역이 저장될 Room ID
    func saveToDB(message: String, forKey roomID: String) {
        if self.realm.objects(ChatModel.self).isEmpty {
            let chatModel = ChatModel()
            let room = Room(value: ["roomID": roomID, "messages": [message]])
            chatModel.rooms.append(room)
            
            try! self.realm.write {
                self.realm.add(chatModel)
            }
        } else {
            let chatModel = self.realm.objects(ChatModel.self).first
            let room = chatModel?.rooms.filter("roomID == '\(roomID)'").first
            
            if room != nil {
                try! self.realm.write {
                    room?.messages.append(message)
                }
            } else {
                let room = Room(value: ["roomID": roomID, "messages": [message]])
                try! self.realm.write {
                    chatModel?.rooms.append(room)
                }
            }
        }
    }
    
    /// 저장된 채팅 내역 불러오기
    ///
    /// - parameters:
    ///     - roomID: 채팅 내역을 불러올 Room ID
    /// - returns: 채팅 내역
    func fetchMessages(forKey roomID: String) -> [String] {
        var messages = [String]()
        
        if let chatModel = self.realm.objects(ChatModel.self).first {
            guard let room = chatModel.rooms.filter("roomID == '\(roomID)'").first else { return messages }
            messages = Array(room.messages)
        }
        
        return messages
    }
    
}
```

```swift
import Foundation

import Starscream

@objc protocol SocketManagerDelegate {
    func didConnected() -> Void
    func didDisconnected() -> Void
    func didReceive(jsonString: String) -> Void
    @objc optional func didReceive(data: Data) -> Void
}

class SocketManager {
    // MARK: - Properties
    
    private let socket = WebSocket(url: URL(string: "ws://13.209.113.221/chat")!)
    var delegate: SocketManagerDelegate?
    
    
    
    // MARK: - Methods
    
    func connect() {
        self.socket.delegate = self
        self.socket.connect()
    }
    
    func disconnect() {
        self.socket.disconnect()
    }
    
    /// 앱에 처음 진입(a100)
    ///
    /// 마지막 메시지, 안읽은 메시지 개수 등 필요 정보 파악
    ///
    /// - parameters:
    ///    - authToken: 로그인시 받은 auth token
    ///    - chattingList: API로 받은 참여중인 채팅방 목록
    func fetchChattingList(to authToken: String) {
        if self.socket.isConnected {
            let header = JSON(["header": ["action": "a100", "authToken": authToken]])
            guard let jsonString = header.rawString(.utf8, options: .prettyPrinted) else { return }
            self.socket.write(string: jsonString)
        }
    }
    
    /// 방에 진입(r100)
    ///
    /// - parameters:
    ///    - roomID: 입장하는 방의 ID
    ///    - authToken: 로그인시 받은 auth token
    func enterRoom(at roomID: String, to authToken: String) {
        if self.socket.isConnected {
            let header = JSON(["header": ["action": "r100", "authToken": authToken],
                               "message": ["roomId": roomID]])
            guard let jsonString = header.rawString(.utf8, options: .prettyPrinted) else { return }
            self.socket.write(string: jsonString)
        }
    }
    
    /// 메시지 전송(m100)
    ///
    /// - parameters:
    ///    - roomID: 방 ID
    ///    - content: 메시지 내용
    ///    - url: 움직이는 이모티콘 URL ?? 멈춰있는 이모티콘 URL
    ///    - imageURL: 멈춰있는 이모티콘 URL
    ///    - adEmoti: 메시지 옆 광고 이모티콘
    ///    - adInfoIdx: 광고 그룹 Idx
    ///    - egroupIdx: 이모티콘 그룹 Idx
    ///    - authToken: 로그인시 받은 auth token
    func send(at roomID: String, content: String, url: String, imageURL: String, adEmoti: String, adInfoIdx: String, egroupIdx: String, to authToken: String) {
        if self.socket.isConnected {
            let header = JSON(["header": ["action": "m100", "authToken": authToken],
                               "message": ["roomId": roomID,
                                           "content": content,
                                           "url": url,
                                           "imageUrl": imageURL,
                                           "adEmoti": adEmoti,
                                           "adInfoIdx": adInfoIdx,
                                           "egroupIdx": egroupIdx]])
            guard let jsonString = header.rawString(.utf8, options: .prettyPrinted) else { return }
            self.socket.write(string: jsonString)
        }
    }
    
}

// MARK: - Extension

extension SocketManager: WebSocketDelegate {
    
    func websocketDidConnect(socket: WebSocketClient) {
        print("소켓 연결됨")
        self.delegate?.didConnected()
    }
    
    func websocketDidDisconnect(socket: WebSocketClient, error: Error?) {
        print("소켓 연결 끊김")
        self.delegate?.didDisconnected()
    }
    
    func websocketDidReceiveMessage(socket: WebSocketClient, text: String) {
        print("텍스트 받음")
        self.delegate?.didReceive(jsonString: text)
    }
    
    func websocketDidReceiveData(socket: WebSocketClient, data: Data) {
        print("데이터 받음")
        self.delegate?.didReceive?(data: data)
    }
    
}
```

```swift
        let url = self.baseURL + "signup"
```
